  FOR ALL USING (auth.uid() = id);

-- Trigger: auto-crear perfil al registrar usuario
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.profiles (id, email)
  VALUES (NEW.id, NEW.email)
  ON CONFLICT DO NOTHING;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

--- hooks/useAuth.ts ---

import { useState, useEffect } from 'react';
import { supabase } from '../lib/supabaseClient';
import { Session } from '@supabase/supabase-js';
import { Profile } from '../types/models';

export function useAuth() {
  const [session, setSession] = useState<Session | null>(null);
  const [profile, setProfile] = useState<Profile | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setSession(session);
      if (session) fetchProfile(session.user.id);
      else setLoading(false);
    });

    const { data: listener } = supabase.auth.onAuthStateChange((_event, session) => {
      setSession(session);
      if (session) fetchProfile(session.user.id);
      else { setProfile(null); setLoading(false); }
    });

    return () => listener.subscription.unsubscribe();
  }, []);

  const fetchProfile = async (userId: string) => {
    const { data } = await supabase
      .from('profiles')
      .select('*')
      .eq('id', userId)
      .single();
    setProfile(data);
    setLoading(false);
  };

  const signIn = async (email: string, password: string) => {
    const { error } = await supabase.auth.signInWithPassword({ email, password });
    if (error) throw error;
  };

  const signOut = async () => {
    await supabase.auth.signOut();
  };

  return { session, profile, loading, signIn, signOut };
}

const BASE_URL = process.env.EXPO_PUBLIC_N8N_BASE_URL;
const TOKEN = process.env.EXPO_PUBLIC_N8N_WEBHOOK_TOKEN;

const headers = {
  'Content-Type': 'application/json',
  'X-Webhook-Token': TOKEN ?? '',
};

// Helper: no bloquear la app si n8n falla
async function safePost(endpoint: string, body: object) {
  try {
    const res = await fetch(`${BASE_URL}/webhook/${endpoint}`, {
      method: 'POST',
      headers,
      body: JSON.stringify(body),
    });
    const data = await res.json();
    console.log(`[n8n] ${endpoint} ✓`, data);
    return { success: true, data };
  } catch (err) {
    console.warn(`[n8n] ${endpoint} ✗ (no bloqueante)`, err);
    return { success: false, error: err };
  }
}

// POST /report-evidence — enviar foto + coordenadas (multipart/form-data)
// IMPORTANTE: este endpoint usa FormData, no JSON
export async function reportEvidence(params: {
  bitacora_id: string;
  custodio_id: string;
  url_imagen: string;
  latitud: number;
  longitud: number;
  timestamp: string;
  contactos?: { remoteJid: string; pushName: string }[];
}) {
  try {
    const formData = new FormData();
    formData.append('bitacora_id', params.bitacora_id);
    formData.append('custodio_id', params.custodio_id);
    formData.append('latitud', String(params.latitud));
    formData.append('longitud', String(params.longitud));
    formData.append('fecha', new Date(params.timestamp).toLocaleDateString('es-MX'));
    formData.append('hora', new Date(params.timestamp).toLocaleTimeString('es-MX'));
    formData.append('estatus', 'en_ruta');
    if (params.contactos) {
      formData.append('contactos', JSON.stringify(params.contactos));
    }
    // Si tienes el archivo local (uri), agrégalo así:
    // formData.append('data', { uri: localUri, name: 'evidencia.jpg', type: 'image/jpeg' } as any);

    const res = await fetch(`${BASE_URL}/webhook/report-evidence`, {
      method: 'POST',
      headers: { 'X-Webhook-Token': TOKEN ?? '' },
      body: formData,
    });
    const data = await res.json();
    console.log('[n8n] report-evidence ✓', data);
    return { success: true, data };
  } catch (err) {
    console.warn('[n8n] report-evidence ✗ (no bloqueante)', err);
    return { success: false, error: err };
  }
}
